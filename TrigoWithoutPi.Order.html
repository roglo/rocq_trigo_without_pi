Definition angle_leb α1 α2 :=
  if (0 ≤? rngl_sin α1)%L then
    if (0 ≤? rngl_sin α2)%L then (rngl_cos α2 ≤? rngl_cos α1)%L
    else true
  else
    if (0 ≤? rngl_sin α2)%L then false
    else (rngl_cos α1 ≤? rngl_cos α2)%L.

Definition angle_ltb α1 α2 :=
  if (0 ≤? rngl_sin α1)%L then
    if (0 ≤? rngl_sin α2)%L then (rngl_cos α2 <? rngl_cos α1)%L
    else true
  else
    if (0 ≤? rngl_sin α2)%L then false
    else (rngl_cos α1 <? rngl_cos α2)%L.
